2. Main App Component (App.js)
javascriptimport React, { useState, useEffect } from 'react';
import zoomSdk from '@zoom/appssdk';
import ParticipantList from './components/ParticipantList';
import SpeakerQueue from './components/SpeakerQueue';
import Timer from './components/Timer';
import Statistics from './components/Statistics';
import './App.css';

function App() {
  const [participants, setParticipants] = useState([]);
  const [queue, setQueue] = useState([]);
  const [currentSpeaker, setCurrentSpeaker] = useState(null);
  const [speakerStats, setSpeakerStats] = useState({});
  const [timeLimit, setTimeLimit] = useState(120); // 2 minutes default

  useEffect(() => {
    initializeZoomApp();
  }, []);

  const initializeZoomApp = async () => {
    try {
      await zoomSdk.config({
        capabilities: ['listParticipants', 'onParticipantChange'],
      });
      
      // Get initial participants
      const { participants } = await zoomSdk.listParticipants();
      setParticipants(participants);
      
      // Listen for participant changes
      zoomSdk.onParticipantChange((event) => {
        setParticipants(event.participants);
      });
    } catch (error) {
      console.error('Error initializing Zoom App:', error);
    }
  };

  const addToQueue = (participant) => {
    if (!queue.find(p => p.userId === participant.userId)) {
      setQueue([...queue, participant]);
    }
  };

  const removeFromQueue = (userId) => {
    setQueue(queue.filter(p => p.userId !== userId));
  };

  const startSpeaking = () => {
    if (queue.length > 0) {
      const speaker = queue[0];
      setCurrentSpeaker(speaker);
      setQueue(queue.slice(1));
      
      // Initialize stats if needed
      if (!speakerStats[speaker.userId]) {
        setSpeakerStats({
          ...speakerStats,
          [speaker.userId]: {
            totalTime: 0,
            instances: 0
          }
        });
      }
    }
  };

  const onTimerComplete = (timeSpoken) => {
    if (currentSpeaker) {
      setSpeakerStats({
        ...speakerStats,
        [currentSpeaker.userId]: {
          totalTime: (speakerStats[currentSpeaker.userId]?.totalTime || 0) + timeSpoken,
          instances: (speakerStats[currentSpeaker.userId]?.instances || 0) + 1
        }
      });
      setCurrentSpeaker(null);
    }
  };

  return (
    <div className="app">
      <h1>Speaker Queue Manager</h1>
      
      <div className="main-layout">
        <ParticipantList 
          participants={participants}
          onAddToQueue={addToQueue}
          speakerStats={speakerStats}
        />
        
        <div className="center-panel">
          <Timer 
            isActive={!!currentSpeaker}
            timeLimit={timeLimit}
            currentSpeaker={currentSpeaker}
            onComplete={onTimerComplete}
          />
          
          <SpeakerQueue 
            queue={queue}
            onRemove={removeFromQueue}
            onStartSpeaking={startSpeaking}
            currentSpeaker={currentSpeaker}
          />
        </div>
        
        <Statistics 
          speakerStats={speakerStats}
          participants={participants}
        />
      </div>
    </div>
  );
}

export default App;
3. Timer Component (components/Timer.js)
javascriptimport React, { useState, useEffect, useRef } from 'react';

function Timer({ isActive, timeLimit, currentSpeaker, onComplete }) {
  const [timeRemaining, setTimeRemaining] = useState(timeLimit);
  const [totalTimeSpoken, setTotalTimeSpoken] = useState(0);
  const intervalRef = useRef(null);

  useEffect(() => {
    if (isActive) {
      setTimeRemaining(timeLimit);
      setTotalTimeSpoken(0);
      
      intervalRef.current = setInterval(() => {
        setTimeRemaining(prev => {
          if (prev <= 1) {
            handleTimerEnd();
            return 0;
          }
          return prev - 1;
        });
        setTotalTimeSpoken(prev => prev + 1);
      }, 1000);
    } else {
      if (intervalRef.current) {
        clearInterval(intervalRef.current);
      }
    }
    
    return () => {
      if (intervalRef.current) {
        clearInterval(intervalRef.current);
      }
    };
  }, [isActive, timeLimit]);

  const handleTimerEnd = () => {
    if (intervalRef.current) {
      clearInterval(intervalRef.current);
    }
    onComplete(totalTimeSpoken);
    // Play sound alert
    playAlert();
  };

  const playAlert = () => {
    const audio = new Audio('data:audio/wav;base64,UklGRiQAAABXQVZFZm10IBAAAAABAAEAQB8AAEAfAAABAAgAZGF0YQAAAAA=');
    audio.play().catch(e => console.log('Could not play alert sound'));
  };

  const formatTime = (seconds) => {
    const mins = Math.floor(seconds / 60);
    const secs = seconds % 60;
    return `${mins}:${secs.toString().padStart(2, '0')}`;
  };

  const getTimerColor = () => {
    const percentage = (timeRemaining / timeLimit) * 100;
    if (percentage > 50) return '#4CAF50';
    if (percentage > 20) return '#FFC107';
    return '#F44336';
  };

  return (
    <div className="timer-container">
      {currentSpeaker && (
        <h2>Current Speaker: {currentSpeaker.displayName}</h2>
      )}
      
      <div className="timer-display" style={{ color: getTimerColor() }}>
        {formatTime(timeRemaining)}
      </div>
      
      <div className="timer-progress">
        <div 
          className="timer-progress-bar"
          style={{ 
            width: `${(timeRemaining / timeLimit) * 100}%`,
            backgroundColor: getTimerColor()
          }}
        />
      </div>
      
      {isActive && (
        <button onClick={() => onComplete(totalTimeSpoken)}>
          End Turn
        </button>
      )}
    </div>
  );
}

export default Timer;
